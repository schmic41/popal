1% ERI = int_repulsion(basis)
 2%
 3% Computes the electron-electron repulsion energy integrals for a given
 4% basis and returns a 4D array with the results.
 5%
 6%   Input:
 7%     basis     1xM list of basis functions with the following fields
 8%        .atom    element number
 9%        .A       cartesian coordinates of center (Bohr radii)
10%        .a       cartesian exponents [ax ay az]
11%        .alpha   list of radial exponents (inverse Bohr radii)
12%        .d       list of contraction coefficients
13%        .N       list of normalization constants
14%
15%   Output:
16%     ERI       MxMxMxM 4D array of repulsion integrals
17
18function ERI = int_repulsion(basis)
19
20nBasis = numel(basis);
21ERI = zeros(nBasis,nBasis,nBasis,nBasis);
22
23% Loop over all unique combinations of 4 basis functions (takes
24% symmetry into account)
25for mu = 1:nBasis
26  for nu = 1:mu
27    for kap = 1:mu
28      for lam = 1:kap
29        
30        % Copy needed information from basis function list, for
31        % faster access
32        A = basis(mu).A;
33        B = basis(nu).A;
34        C = basis(kap).A;
35        D = basis(lam).A;
36        alpha = basis(mu).alpha;
37        beta =  basis(nu).alpha;
38        gamma = basis(kap).alpha;
39        delta = basis(lam).alpha;
40        
41        dNmu = basis(mu).d.*basis(mu).N;
42        dNnu = basis(nu).d.*basis(nu).N;
43        dNlam = basis(kap).d.*basis(kap).N;
44        dNsig = basis(lam).d.*basis(lam).N;
45        
46        a = basis(mu).a;
47        b = basis(nu).a;
48        c = basis(kap).a;
49        d = basis(lam).a;
50        
51        % Calculate repulsion integral <mu,nu|lam,sig> by contracting
52        % primitive integrals [k,l|n,o]
53        eri_ = 0;
54        for k = 1:numel(dNmu)
55          for l = 1:numel(dNnu)
56            for n = 1:numel(dNlam)
57              for o = 1:numel(dNsig)
58                eri_ = eri_ + ...
59                    dNmu(k)*dNnu(l)*dNlam(n)*dNsig(o) * ...
60                      eri_primitive_fast(A,B,C,D,...
61                           alpha(k),beta(l),gamma(n),delta(o),...
62                           a,b,c,d);
63              end
64            end
65          end
66        end
67        
68        % Utilize symmetry properties of two-electron integrals
69        ERI(mu,nu,kap,lam) = eri_;
70        ERI(nu,mu,kap,lam) = eri_;
71        ERI(mu,nu,lam,kap) = eri_;
72        ERI(nu,mu,lam,kap) = eri_;
73        
74        ERI(kap,lam,mu,nu) = eri_;
75        ERI(kap,lam,nu,mu) = eri_;
76        ERI(lam,kap,mu,nu) = eri_;
77        ERI(lam,kap,nu,mu) = eri_;
78        
79      end
80    end
81  end
82end
